OnStart:
	announce "- Duas equipes entraram na [ Arena 3 - niff 25ros03_2 ]. -", bc_yellow;
	set $@KvM82BG_Victory, 0;
	
	// Criar partida no sistema de ranking ANTES do warp (para rastrear desertores durante preparo)
	donpcevent "Arena7x7_Ranking_03_Hook::OnMatchCreate";
	
	// Remover buffs de todos em quiz_02 antes de teleportar
	getmapunits BL_PC, "quiz_02", .@units;
	for (set .@i, 0; .@i < getarraysize(.@units); set .@i, .@i + 1) {
		if (isloggedin(.@units[.@i])) {
			attachrid .@units[.@i];
			sc_end SC_ALL;
			setoption 0x40 | Option_Invisible | Option_Xmas, 0;
			sc_end SC_FIGHTINGSPIRIT;
			sc_end SC_LUXANIMA;
			sc_end SC_GIANTGROWTH;
			sc_end SC_STONEHARDSKIN;
			sc_end SC_VITALITYACTIVATION;
			sc_end SC_ABUNDANCE;
			sc_end SC_CRUSHSTRIKE;
			sc_end SC_REFRESH;
			sc_end SC_MILLENNIUMSHIELD;

			sc_end SC_SAVAGE_STEAK;
			sc_end SC_COCKTAIL_WARG_BLOOD;

			sc_end SC_SIROMA_ICE_TEA;
			sc_end SC_DROCERA_HERB_STEAMED;
			sc_end SC_PUTTI_TAILS_NOODLES;
			sc_end SC_STOMACHACHE;
			sc_end SC_VITATA_500;
			sc_end SC_EXTRACT_SALAMINE_JUICE;
		}
	}
	
	donpcevent "Arena_AutoRestock::OnRestock";
	atcommand "@repairall";
	// Warp Teams
	bg_warp $@KvM82BG_id1,"25ros03_2",46,79;
	bg_warp $@KvM82BG_id2,"25ros03_2",153,79;

	if ($@guildsInArena > 0) {
        $@guildsInArena--;
    }
	if ($@guildsInArena > 0) {
        $@guildsInArena--;
    }

	// setwall "25ros03_2",229,113,8,4,0,"batc01wall_ae01";
	// setwall "25ros03_2",113,113,8,4,0,"batc01wall_ae02";

	set .Croix_Score, 0;

	set .Croix_Count, bg_get_data($@KvM82BG_id2, 0);

	// O placar mostra: Score Azul | Vivos Azul vs Vivos Vermelho | Score Vermelho
	// bg_updatescore usa os valores como placar esquerdo e direito
	// Mas precisamos mostrar jogadores vivos, ent?o usaremos Count
	bg_updatescore "25ros03_2",.Guillaume_Count,.Croix_Count;
	
	initnpctimer;
	end;

OnTimer25000:

	set .@blue_check, bg_get_data($@KvM82BG_id1, 0);
	set .@red_check, bg_get_data($@KvM82BG_id2, 0);
	
	if( (.@blue_check < $@jogadoresblue) || (.@red_check < $@jogadoresred) )
	{
		set $@KvM82BG_Victory, 3;
		set $@KvM82BG, 3;
		
		set .@missing_blue, $@jogadoresblue - .@blue_check;
		set .@missing_red, $@jogadoresred - .@red_check;
		
		if (.@missing_blue > 0 && .@missing_red > 0) {
			mapannounce "25ros03_2","- Partida cancelada: jogadores desconectaram de ambos os times! -", bc_yellow;
		} else if (.@missing_blue > 0) {
			mapannounce "25ros03_2","- Partida cancelada: " + .@missing_blue + " jogador(es) do Time Azul desconectou! -", bc_yellow;
		} else {
			mapannounce "25ros03_2","- Partida cancelada: " + .@missing_red + " jogador(es) do Time Vermelho desconectou! -", bc_yellow;
		}
		
		donpcevent "Arena7x7_Ranking_03_Hook::OnMatchCancel";
		stopnpctimer;
		sleep 2000;
		donpcevent "KvM82_BG::OnReset";  // Usar OnReset para destruir BG imediatamente

		if ($@RedGuildName$ == "") set .@redname$, "Red Team"; else set .@redname$, $@RedGuildName$;
		
		// Detectar qual time perdeu jogadores
		set .@missing_blue, $@jogadoresblue - .@blue_count_real;
		set .@missing_red, $@jogadoresred - .@red_count_real;
		
		if (.@missing_blue > 0 && .@missing_red > 0) {
			mapannounce "25ros03_2","- Partida cancelada: jogadores desconectaram de ambos os times! -", bc_yellow;
			mapannounce "25ros03_2","- " + .@bluename$ + ": " + .@blue_count_real + "/" + $@jogadoresblue + " | " + .@redname$ + ": " + .@red_count_real + "/" + $@jogadoresred + " -", bc_yellow;

			mapannounce "25ros03_2","- Partida cancelada: " + .@missing_blue + " jogador(es) da Guild " + .@bluename$ + " desconectou! -", bc_yellow;
			mapannounce "25ros03_2","- " + .@bluename$ + ": " + .@blue_count_real + "/" + $@jogadoresblue + " players presentes -", bc_yellow;
		} else {
			mapannounce "25ros03_2","- Partida cancelada: " + .@missing_red + " jogador(es) da Guild " + .@redname$ + " desconectou! -", bc_yellow;
			mapannounce "25ros03_2","- " + .@redname$ + ": " + .@red_count_real + "/" + $@jogadoresred + " players presentes -", bc_yellow;
		}
		
		// Cancelar a partida no sistema de ranking (se foi criada)
		donpcevent "Arena7x7_Ranking_03_Hook::OnMatchCancel";
		
		stopnpctimer;
		sleep 2000;
		donpcevent "KvM82_BG::OnStop";
		end;

	set .Croix_Count, .@red_count_real;

       set $spectator_count9, $spectator_count9 - 1;
    }
	if ($spectator_count9 > 0) {
       set $spectator_count9, $spectator_count9 - 1;

		arena7x7_restoreall($@arena3_match_id);
	}
	donpcevent "Arena_AutoRestock::OnRestock3";
	sleep 500;
	// Teleportar membros do BG (vivos)

	if ($@KvM82BG_id2) bg_warp $@KvM82BG_id2,"izlude",128,135;

	donpcevent "KvM82_BG_Out::OnBegin";
	end;

OnReset:
	donpcevent "Arena7x7_Ranking_03_Hook::OnMatchCancel";
	stopnpctimer;
	stopnpctimer "KvM82_BG_Out";
	set .Croix_Count, 0;
	set .Guillaume_Count, 0;
	set .Croix_Score, 0;
	set .Guillaume_Score, 0;
	set $@KvM82BG_Victory, 0;

	if( $@KvM80BG == 3 ) {
		bg_warp $@KvM80BG_id1,"izlude",128,135;
		bg_warp $@KvM80BG_id2,"izlude",128,135;
		mapwarp "ros2024","izlude",128,135;
	}
	// Destruir BG
	if( $@KvM82BG_id1 ) { bg_destroy $@KvM82BG_id1; set $@KvM82BG_id1, 0; }
	if( $@KvM82BG_id2 ) { bg_destroy $@KvM82BG_id2; set $@KvM82BG_id2, 0; }
	sleep 2000;
	maprespawnguildid "25ros03_2",0,3; // Just in case someone else
	bg_updatescore "25ros03_2",0,0;
	sleep 2000;
	set $@KvM82BG, 0;
	donpcevent "KvM82_BG::OnReadyCheck"; // Maybe a game is ready to start
	end;

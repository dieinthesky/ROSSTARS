		JobLevel = 70;

GiveCardTokens:
	.@tokens = callsub(GetInitialCardTokens);
	if (.@tokens <= 0)
		return;

	.@current = countitem(1000196);
	if (.@current > 0)
		delitem 1000196,.@current;

	if (!checkweight(1000196,.@tokens)) {
		message strcharinfo(0),"Libere peso no inventario para receber seus tokens de cartas.";
		return;
	}

	getitembound 1000196,.@tokens,BOUND_ACCOUNT;
	set #ROS2026_CARDTOKEN_CLASS, Class;
	set #ROS2026_CARDTOKEN_START, .@tokens;
	message strcharinfo(0),"Voce recebeu "+.@tokens+" tokens de cartas da sua classe.";
	return;

GetInitialCardTokens:

//===== rAthena Script =======================================
//= Mestre das Montarias (3rd Trans) - turbo_room - ROS 7X7
//= Texto em ASCII sem acentos: evita '?' no jogo se o .txt for salvo como UTF-8.
//= Variavel permanente: #ROS2026_MOUNT_PREF (10=carrinho 11=mado 12=ambos; ver L_SavePref)
//===== Mapa: turbo_room,95,123
//============================================================
turbo_room,95,123,4	script	Mestre das Montarias#ros2026	105,{

	if (ismounting()) {
		mes "[Mestre das Montarias]";
		mes "Para preparacao competitiva no ROS 7X7, retire primeiro a montaria premium.";
		mes "Assim evitamos conflito com montarias de classe.";
		close;
	}

	set .@c, Class;
	if (.@c != 4060 && .@c != 4061 && .@c != 4062 && .@c != 4063 && .@c != 4064 && .@c != 4065
		&& .@c != 4073 && .@c != 4074 && .@c != 4075 && .@c != 4076 && .@c != 4077 && .@c != 4078 && .@c != 4079) {
		mes "[Mestre das Montarias]";
		mes "Este posto atende apenas classes de terceira transcendente homologadas para o servidor.";
		mes "Sua classe atual nao consta na lista de atendimento.";
		close;
	}

	mes "[Mestre das Montarias]";
	mes "Bem-vindo. Servidor competitivo ROS 7X7: padronizamos montarias e companheiros para partidas serias.";
	mes "As habilidades exigidas precisam estar aprendidas e com ponto investido, conforme a regra do emulador.";
	mes "Como posso ajudar?";
	next;
	switch(select("Aplicar ou trocar montaria agora:Registrar preferencia (Curandeira):Regras Carrinho e Mado:Encerrar")) {
	case 1:
		callsub L_Apply;
		end;
	case 2:
		callsub L_SavePref;
		end;
	case 3:
		mes "[Mestre das Montarias]";
		mes "Mecanico: neste servidor voce pode usar ^0000ffCarrinho + Mado juntos^000000 pela opcao dedicada no menu.";
		mes "Se escolher apenas Carrinho ou apenas equipamento Mado, o outro modo e desligado para evitar estado misto indesejado.";
		next;
		mes "[Mestre das Montarias]";
		mes "Ranger: Falcao e Warg nunca ficam ativos juntos. A escolha reflete seu preparo tatico no ROS 7X7.";
		close;
	default:
		mes "[Mestre das Montarias]";
		mes "Boa sorte nas partidas.";
		close;
	}

L_Apply:
		switch (Class) {
		case 4060:
			if (!getskilllv("RK_DRAGONTRAINING")) {
				mes "[Mestre das Montarias]";
				mes "E obrigatorio possuir ^0000ffTreinamento com Dragao^000000 (RK_DRAGONTRAINING) com pelo menos 1 nivel.";
				close;
			}
			if (checkdragon()) {
				mes "[Mestre das Montarias]";
				mes "Voce ja esta com o dragao convocado.";
				close;
			}
			setdragon;
			mes "[Mestre das Montarias]";
			mes "Dragao liberado. Mantenha-o sob controle durante o combate.";
			callsub L_AskSaveNow, 20;
			close;

		case 4073:
			if (!getskilllv("KN_RIDING")) {
				mes "[Mestre das Montarias]";
				mes "Aprenda ^0000ffMontaria^000000 (KN_RIDING) antes de solicitar a montaria.";
				close;
			}
			if (checkriding()) {
				mes "[Mestre das Montarias]";
				mes "Voce ja utiliza a montaria de grifo.";
				close;
			}
			setriding;
			mes "[Mestre das Montarias]";
			mes "Montaria de grifo aplicada com sucesso.";
			callsub L_AskSaveNow, 30;
			close;

		case 4062:
			callsub L_RangerApply;
			return;

		case 4064:
			callsub L_MechApply;
			return;

		case 4078:
			if (!getskilllv("MC_PUSHCART")) {
				mes "[Mestre das Montarias]";
				mes "Desbloqueie ^0000ffEmpurrar Carrinho^000000 para usar este servico.";
				close;
			}
			if (checkcart()) {
				mes "[Mestre das Montarias]";
				mes "Voce ja possui carrinho ativo.";
				close;
			}
			setcart;
			mes "[Mestre das Montarias]";
			mes "Carrinho aplicado.";
			callsub L_AskSaveNow, 40;
			close;

		case 4061:
		case 4063:
		case 4065:
		case 4074:
		case 4075:
		case 4076:
		case 4077:
		case 4079:
			mes "[Mestre das Montarias]";
			mes "Sua classe nao utiliza os mesmos recursos de montaria ou carrinho que Cavaleiro Runico,";
			mes "Guarda Real, Ranger, Mecanico ou Genetico. Nada a disponibilizar por aqui.";
			close;
		}
		return;

L_RangerApply:
		mes "[Mestre das Montarias]";
		mes "Ranger: escolha um unico modo: Falcao ou Warg montado, conforme seu planejamento para o ROS 7X7.";
		next;
		switch(select("Falcao:Warg montado:Cancelar")) {
		case 1:
			if (!getskilllv("HT_FALCON")) {
				mes "[Mestre das Montarias]";
				mes "Invista ao menos 1 ponto em ^0000ffDomesticacao de Falcao^000000.";
				close;
			}
			if (checkwug()) {
				setoption (OPTION_WUG|OPTION_WUGRIDER), 0;
			}
			if (checkfalcon()) {
				mes "[Mestre das Montarias]";
				mes "O falcao ja esta ativo.";
				close;
			}
			setfalcon;
			mes "[Mestre das Montarias]";
			mes "Falcao liberado.";
			callsub L_AskSaveNow, 1;
			close;
		case 2:
			if (!getskilllv("RA_WUGMASTERY") || !getskilllv("RA_WUGRIDER")) {
				mes "[Mestre das Montarias]";
				mes "Sao necessarios ^0000ffDominio sobre Warg^000000 e";
				mes "^0000ffMontaria em Warg^000000, cada um com pelo menos 1 nivel.";
				close;
			}
			setfalcon 0;
			setoption OPTION_WUGRIDER, 1;
			mes "[Mestre das Montarias]";
			mes "Warg montado. Bom desempenho na arena.";
			callsub L_AskSaveNow, 2;
			close;
		default:
			mes "[Mestre das Montarias]";
			mes "Quando precisar, estarei a disposicao.";
			close;
		}

L_MechApply:
		mes "[Mestre das Montarias]";
		mes "Mecanico: escolha Carrinho, equipamento Mado, ou ^0000ffambos ao mesmo tempo^000000.";
		next;
		switch(select("Carrinho:Equipamento Mado:Carrinho e Mado juntos:Cancelar")) {
		case 1:
			if (!getskilllv("MC_PUSHCART")) {
				mes "[Mestre das Montarias]";
				mes "Aprenda ^0000ffEmpurrar Carrinho^000000 para continuar.";
				close;
			}
			if (checkmadogear())
				setmadogear 0;
			if (checkcart()) {
				mes "[Mestre das Montarias]";
				mes "Carrinho ja esta em uso.";
				close;
			}
			setcart;
			mes "[Mestre das Montarias]";
			mes "Carrinho ativo. O Mado foi desativado caso estivesse ligado.";
			callsub L_AskSaveNow, 10;
			close;
		case 2:
			if (!getskilllv("NC_MADOLICENCE")) {
				mes "[Mestre das Montarias]";
				mes "Obtenha ^0000ffLicenca Mado^000000 antes de prosseguir.";
				close;
			}
			if (checkcart())
				setcart 0;
			if (checkmadogear()) {
				mes "[Mestre das Montarias]";
				mes "Voce ja esta operando o equipamento Mado.";
				close;
			}
			setmadogear;
			mes "[Mestre das Montarias]";
			mes "Equipamento Mado ativo. Carrinho foi removido se necessario.";
			callsub L_AskSaveNow, 11;
			close;
		case 3:
			if (!getskilllv("MC_PUSHCART") || !getskilllv("NC_MADOLICENCE")) {
				mes "[Mestre das Montarias]";
				mes "Para os dois modos juntos e necessario ^0000ffEmpurrar Carrinho^000000 e ^0000ffLicenca Mado^000000.";
				close;
			}
			if (!checkcart())
				setcart;
			if (!checkmadogear())
				setmadogear;
			mes "[Mestre das Montarias]";
			mes "Carrinho e Mado ativos. Boa preparacao na arena.";
			callsub L_AskSaveNow, 12;
			close;
		default:
			mes "[Mestre das Montarias]";
			mes "Operacao cancelada.";
			close;
		}

L_SavePref:
		switch (Class) {
		case 4062:
			mes "[Mestre das Montarias]";
			mes "Defina qual opcao a Curandeira devera priorizar apos o restabelecimento (ROS 7X7).";
			next;
			switch(select("Falcao (registro 1):Warg (registro 2):Limpar registro (0)")) {
			case 1:
				if (!getskilllv("HT_FALCON")) {
					mes "[Mestre das Montarias]";
					mes "Sem a skill de Falcao nao e possivel registrar essa preferencia.";
					close;
				}
				set #ROS2026_MOUNT_PREF, 1;
				break;
			case 2:
				if (!getskilllv("RA_WUGMASTERY") || !getskilllv("RA_WUGRIDER")) {
					mes "[Mestre das Montarias]";
					mes "Complete as habilidades de Warg antes de registrar esta opcao.";
					close;
				}
				set #ROS2026_MOUNT_PREF, 2;
				break;
			default:
				set #ROS2026_MOUNT_PREF, 0;
			}
			mes "[Mestre das Montarias]";
			mes "Registro salvo (#ROS2026_MOUNT_PREF). A Curandeira aplicara conforme o script do servidor.";
			close;

		case 4064:
			mes "[Mestre das Montarias]";
			mes "Qual modo a Curandeira deve aplicar: Carrinho, Mado, ou ambos juntos?";
			next;
			switch(select("Carrinho (10):Mado (11):Carrinho + Mado (12):Limpar (0)")) {
			case 1:
				if (!getskilllv("MC_PUSHCART")) {
					mes "[Mestre das Montarias]";
					mes "Empurrar Carrinho e requisito para salvar o carrinho.";
					close;
				}
				set #ROS2026_MOUNT_PREF, 10;
				break;
			case 2:
				if (!getskilllv("NC_MADOLICENCE")) {
					mes "[Mestre das Montarias]";
					mes "Licenca Mado e requisito para salvar o Mado.";
					close;
				}
				set #ROS2026_MOUNT_PREF, 11;
				break;
			case 3:
				if (!getskilllv("MC_PUSHCART") || !getskilllv("NC_MADOLICENCE")) {
					mes "[Mestre das Montarias]";
					mes "E preciso ter as duas habilidades para registrar carrinho e Mado juntos.";
					close;
				}
				set #ROS2026_MOUNT_PREF, 12;
				break;
			default:
				set #ROS2026_MOUNT_PREF, 0;
			}
			mes "[Mestre das Montarias]";
			mes "Preferencia registrada (#ROS2026_MOUNT_PREF). Opcao 12 aplica os dois sem remover um pelo outro.";
			close;

		case 4060:
			if (!getskilllv("RK_DRAGONTRAINING")) {
				mes "[Mestre das Montarias]";
				mes "Sem Treinamento com Dragao nao ha preferencia valida a registrar.";
				close;
			}
			set #ROS2026_MOUNT_PREF, 20;
			mes "[Mestre das Montarias]";
			mes "Registrado: Dragao (codigo 20).";
			close;

		case 4073:
			if (!getskilllv("KN_RIDING")) {
				mes "[Mestre das Montarias]";
				mes "Sem a skill Montaria nao ha registro aplicavel.";
				close;
			}
			set #ROS2026_MOUNT_PREF, 30;
			mes "[Mestre das Montarias]";
			mes "Registrado: Grifo (codigo 30).";
			close;

		case 4078:
			if (!getskilllv("MC_PUSHCART")) {
				mes "[Mestre das Montarias]";
				mes "E necessario possuir Empurrar Carrinho para registrar.";
				close;
			}
			set #ROS2026_MOUNT_PREF, 40;
			mes "[Mestre das Montarias]";
			mes "Registrado: Carrinho para Genetico (codigo 40).";
			close;

		default:
			mes "[Mestre das Montarias]";
			mes "Para sua classe nao existe opcao adicional de registro neste terminal.";
			close;
		}
		return;

L_AskSaveNow:
	mes "[Mestre das Montarias]";
	mes "Deseja salvar escolha?";
	next;
	if (select("Sim, salvar para Curandeira:Agora nao") == 1) {
		set #ROS2026_MOUNT_PREF, getarg(0);
		mes "[Mestre das Montarias]";
		mes "Escolha salva com sucesso para a Curandeira.";
	} else {
		// Evita manter preferencia antiga (ex.: Mado) apos trocar para outra opcao.
		// Sem salvamento explicito, a Curandeira nao deve aplicar montaria automaticamente.
		set #ROS2026_MOUNT_PREF, 0;
		mes "[Mestre das Montarias]";
		mes "Sem problemas. A escolha nao foi salva e o registro anterior foi limpo.";
	}
	return;

OnInit:
	initnpctimer;
	end;
OnTimer1000:
	showscript "Montarias ROS 7X7";
	setnpctimer 0;
	end;
}

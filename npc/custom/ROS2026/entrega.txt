turbo_room,109,110,4	script	Maquina de Restoque	564,{

	// variaveis permanentes por personagem:
	// StuffsSaved = 0/1, Stuffs_id[], Stuffs_amt[]
	// Salva consumiveis + ETC (exceto cartas e item 1000196): F_ROS2026_IsRestockConsumable em healer.txt

	mes "[Maquina de Restoque]";
	if (StuffsSaved) {
		mes "Voce ja possui uma lista de suprimentos salva.";
		next;

		// Jogador ja tem lista salva
		// 1 - Sobrescrever com itens atuais
		// 2 - Apagar lista salva
		// 3 - Resetar Zenys para 20.000.000
		// 4 - Cancelar
		.@opt = select(
			"Sobrescrever lista salva com os itens atuais",
			"Apagar lista salva",
			"Resetar Zenys para 20.000.000",
			"Cancelar"
		);

		if (.@opt == 1) {
			mes "[Maquina de Restoque]";
			mes "Esta acao ira apagar a lista de suprimentos salva e registrar a lista atual do seu inventario.";
			next;
			if (select("Confirmar sobrescrita:Cancelar") == 2) close;
			goto L_SaveCurrent;
		}

		if (.@opt == 2) {
			mes "[Maquina de Restoque]";
			mes "Esta acao ira apagar definitivamente a sua lista de suprimentos salva.";
			next;
			if (select("Apagar lista:Cancelar") == 2) close;

			deletearray Stuffs_id[0], getarraysize(Stuffs_id);
			deletearray Stuffs_amt[0], getarraysize(Stuffs_amt);
			StuffsSaved = 0;

			mes "[Maquina de Restoque]";
			mes "A lista de suprimentos foi removida com sucesso.";
			close;
		}

		if (.@opt == 3)
			goto L_ResetZeny;

		close;
	} else {
		mes "Voce ainda nao possui uma lista de suprimentos salva.";
		next;

		// Jogador nao tem lista salva
		// 1 - Salvar stuffs atuais
		// 2 - Resetar Zenys
		// 3 - Cancelar
		.@opt = select(
			"Salvar meus suprimentos atuais",
			"Resetar Zenys para 20.000.000",
			"Cancelar"
		);

		if (.@opt == 1)
			goto L_SaveCurrent;
		if (.@opt == 2)
			goto L_ResetZeny;

		close;
	}

L_SaveCurrent:
	getinventorylist;

	if (getarraysize(@inventorylist_id) == 0) {
		mes "[Maquina de Restoque]";
		mes "Seu inventario esta vazio. Nao ha itens para salvar.";
		close;
	}

	.@restock_count = 0;
	for (.@i = 0; .@i < getarraysize(@inventorylist_id); .@i++) {
		if (@inventorylist_equip[.@i]) continue;
		if (callfunc("F_ROS2026_IsRestockConsumable", @inventorylist_id[.@i]))
			.@restock_count++;
	}
	if (.@restock_count == 0) {
		mes "[Maquina de Restoque]";
		mes "Nao ha itens elegiveis (consumiveis ou ETC, nao equipados) no inventario para salvar.";
		close;
	}

	mes "[Maquina de Restoque]";
	mes "A lista abaixo contem consumiveis e itens ETC nao equipados (exceto cartas e item 1000196):";
	for (.@i = 0; .@i < getarraysize(@inventorylist_id); .@i++) {
		if (@inventorylist_equip[.@i]) continue;
		if (!callfunc("F_ROS2026_IsRestockConsumable", @inventorylist_id[.@i])) continue;
		mes "- ID "+@inventorylist_id[.@i]+"  qtd: "+@inventorylist_amount[.@i];
	}
	next;

	if (select("Salvar lista:Cancelar") == 2) close;

	deletearray Stuffs_id[0], getarraysize(Stuffs_id);
	deletearray Stuffs_amt[0], getarraysize(Stuffs_amt);

	for (.@i = 0; .@i < getarraysize(@inventorylist_id); .@i++) {
		if (@inventorylist_equip[.@i]) continue;
		if (!callfunc("F_ROS2026_IsRestockConsumable", @inventorylist_id[.@i])) continue;
		.@idx = getarraysize(Stuffs_id);
		Stuffs_id[.@idx] = @inventorylist_id[.@i];
		Stuffs_amt[.@idx] = @inventorylist_amount[.@i];
	}

	StuffsSaved = 1;

	mes "[Maquina de Restoque]";
	mes "Sua lista de suprimentos foi salva com sucesso.";
	close;

L_ResetZeny:
	.@limit = 20000000;
	if (Zeny < .@limit)
		Zeny = .@limit;

	close;

OnInit:
	initnpctimer;
	end;
OnTimer1000:
	showscript "Maquina de Restoque";
	setnpctimer 0;
	end;
}
